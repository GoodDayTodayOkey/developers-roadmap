```

  field: number;
  getThis: // указать тип;
}

class Something {
  public field: number;
  constructor(field: number) { this.field = field; }
  public getThis() /* указать тип */ {
    // вернуть контекст;
  }
}

const instance: ISomething = new Something(0);
const result: ISomething = instance.getThis();
```
